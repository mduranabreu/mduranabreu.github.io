  <script>

    let viewer = null;

    let model = null;


    function getQueryParam(name) {

      const urlParams = new URLSearchParams(window.location.search);

      return urlParams.get(name);

    }


    function getFileExtension(filename) {

      return filename.split('.').pop().toLowerCase();

    }


    function getFormat(ext) {

      const map = {

        xyz: "xyz",

        mol: "mol",

        sdf: "sdf",

        pdb: "pdb"

      };

      return map[ext] || null;

    }


    function getFileName(path) {

      return path.split('/').pop();

    }


    async function loadMolecule() {

      const file = getQueryParam("file");


      if (!file) {

        alert("No se ha indicado ningún archivo. Usa ?file=moleculas/archivo.xyz");

        return;

      }


      const ext = getFileExtension(file);

      const format = getFormat(ext);


      if (!format) {

        alert("Formato no soportado: " + ext);

        return;

      }


      document.getElementById("title").innerText = getFileName(file);


      const element = document.getElementById("viewer");

      viewer = $3Dmol.createViewer(element, { backgroundColor: "white" });


      try {

        const response = await fetch(file);

        const data = await response.text();


        model = viewer.addModel(data, format);


        viewer.setStyle({}, { stick: {}, sphere: { scale: 0.28 } });

        viewer.zoomTo();

        viewer.render();

      } catch (error) {

        console.error("Error cargando la molécula:", error);

        alert("No se pudo cargar el archivo molecular.");

      }

    }


    function setBallStick() {

      viewer.setStyle({}, { stick: {}, sphere: { scale: 0.28 } });

      viewer.render();

    }


    function setStick() {

      viewer.setStyle({}, { stick: { radius: 0.18 } });

      viewer.render();

    }


    function setSphere() {

      viewer.setStyle({}, { sphere: {} });

      viewer.render();

    }


    function resetView() {

      viewer.zoomTo();

      viewer.render();

    }


    loadMolecule();

  </script>
